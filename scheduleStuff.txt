                        '<p class="date"></p>' +

                        '<p class="topic"></p>' +

        item.find('.topic').text(topic);
        item.find('.task').text(task);

        $('.information-container').append(item);

        // Add a separator between each item
        if ($('.information-container').children().length > 1) {
          $('.information-container').children().not(':last-child').after('<div class="separator"></div>');
        }
      }
